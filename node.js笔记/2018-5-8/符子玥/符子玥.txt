5.请用express框架搭建一个hello world服务器
答：var express = require("express");
    var app = express();
    app.listen(3000,function(){
         console.log("服务器已启动")
         console.log("localhost:3000")
    });
    app.use(function(req,res){
         res.send("hello world")
    });
